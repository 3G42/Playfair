void preenche_palavra(char palavra[])
{
	int i, j;
	gotoxy(9,12);
	scanf("%s", palavra);
	for(i=0;palavra[i] != '\0'; i++)
	{
		palavra[i] = toupper(palavra[i]);
	}
    for(i=0; i < strlen(palavra); i++)
        while(palavra[i] == ' ')
		{
            for(j=i; j < strlen(palavra); j++)
            {
                palavra[j] = palavra[j+1];
        	}
        }
	
}
